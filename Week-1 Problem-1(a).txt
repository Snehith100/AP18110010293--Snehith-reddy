#include <stdio.h>

int main()
{
    int n;
    
    printf("enter the size : ");
    scanf("%d",&n);
   
    printf("enter the binary string \n");
    int z,o,i=0;

    for(i=0;i<n;i++){
        int temp;
        printf("enter binary digit %d : ",i+1);
        scanf("%d",&temp);
        if(temp==0){
            z+=1;
        }
        else{
            o+=1;
        }
    }
  
    if(z%2==0 && o%2==0){
        printf("Given binary string contains even number of 0's and 1's");
    }
    else{
        printf("Given binary string does not contain even number of 0's and 1's ");
    }
    return 0;
}
